93 void loop() {
94  if(!stringComplete)
95  return;
96 
97  if(writingToFlash) {
98  if(String("!ABORTWRITEFLASHMEMORY!").equals(inputString)) {
99  writingToFlash = false;
100  Serial.println("\nAbort writing to flash memory...");
101  }
102  else if(!max1464.writeHexLineToFlashMemory(inputString)) {
103  Serial.print("\nIllegal line ");
104  Serial.println(inputString);
105  }
106  else {
107  Serial.print(".");
108  hexLinesWritten++;
109  if(hexLinesWritten>80) {
110  hexLinesWritten = 0;
111  Serial.println();
112  }
113  if(max1464.hasEOFBeenReached()) {
114  writingToFlash = false;
115  Serial.println();
116  }
117  }
118  clearInputString();
119  return;
120  }
121 
122  inputString.toUpperCase();
123  inputString = strtok((char *)inputString.c_str(), " ");
124 
125  if(String("IDEN").startsWith(inputString)) {
126  printIden();
127  }
128  else if(String("RFW").startsWith(inputString)) {
129  char *partition_cp = strtok(NULL, " ");
130  uint8_t partition = 0;
131  if(partition_cp != NULL) {
132  partition = atoi(partition_cp);
133  }
134  max1464.readFlashPartition((FLASH_PARTITION)partition);
135  }
136  else if(String("HALTCPU").startsWith(inputString)) {

139  }
140  else if(String("RESETCPU").startsWith(inputString)) {

143  }
144  else if(String("RP").startsWith(inputString)) {
145  char *port_cp = strtok(NULL, " ");

147  uint8_t port = atoi(port_cp);
148  Serial.print("CPU port ");

178  }
179  else {
